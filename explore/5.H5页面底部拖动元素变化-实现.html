<template>

                    @touchend="endTouch"></div>

                <div class="org-list">

            <van-popup v-model:show="show" round position="bottom" :overlay-style="{ backgroundColor: 'transparent' }"

                    <div class="title">{{ activeRow.orgName }}</div>

                </div>
            </van-popup>
        </div>
    </PageView>
</template>

        tel: '025-154266853',
        type: 'contact',
        lat: '',
        lng: ''
    }
])

const activeRow = ref<CommonObj>({})
const onOpenOrg = (row: CommonObj) => {
    show.value = true
    activeRow.value = row
}

const containerRef = ref()
const targetRef = ref()
const minHeight = 200
let isDragging = false
let startY: number, initHeight: number
const startTouch = (el: TouchEvent) => {
    startY = el.touches[0].clientY
    isDragging = true
    initHeight = targetRef.value.clientHeight
}
const moveTouch = (el: TouchEvent) => {
    if (!isDragging) return
    const tochY = el.touches[0].clientY
    const diffY = tochY - startY
    const newHeight = initHeight - diffY
    let maxHeight = containerRef.value.clientHeight - 80
    targetRef.value.style.height = Math.min(maxHeight, Math.max(minHeight, newHeight)) + 'px'
    el.preventDefault()
}
const endTouch = (el: TouchEvent) => {
    isDragging = false
}
</script>
